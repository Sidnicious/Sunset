def fit_and_plot(data, f, hint=None):
    fit = curve_fit(f, *data, maxfev=100000, p0=hint)[0]
    fit_vals = f(data[0], *fit)
    plt.plot(data[0], data[1], 'r', data[0], fit_vals, 'b')
    print("fit:", fit)
    print("R^2:", np.corrcoef(data[1], fit_vals)[0,1] ** 2)
    return fit

def f(x, a, b, c, d, e):
    logx = np.log(x)
    return a * logx ** 4 + b * logx ** 3 + c * logx ** 2 + d * logx + e
